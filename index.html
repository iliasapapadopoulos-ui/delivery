let currentAddress = ""; // Εναλλαγή Φωνής / Πλήκτρων function setMode(mode) { const btnVoice = document.getElementById('btn-voice'); const btnKeys = document.getElementById('btn-keys'); const voiceUI = document.getElementById('voice-ui'); const keysUI = document.getElementById('keys-ui'); if (mode === 'voice') { btnVoice.classList.add('active'); btnKeys.classList.remove('active'); voiceUI.style.display = 'block'; keysUI.style.display = 'none'; } else { btnVoice.classList.remove('active'); btnKeys.classList.add('active'); voiceUI.style.display = 'none'; keysUI.style.display = 'block'; } } // ATM Style μορφοποίηση (max 99.99) function formatAtm(input) { let digits = input.value.replace(/\D/g, ''); if (!digits) { input.value = "0.00"; return; } let value = parseInt(digits, 10); if (value > 9999) value = 9999; // Όριο 99.99 let formatted = (value / 100).toFixed(2); input.value = formatted; } // Αναγνώριση Φωνής (Web Speech API) function startSpeech() { const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { alert("Η αναγνώριση φωνής δεν υποστηρίζεται."); return; } const recognition = new SpeechRecognition(); recognition.lang = 'el-GR'; const btn = document.querySelector('.voice-active-btn'); btn.innerText = "Ακούω..."; recognition.onresult = function(event) { currentAddress = event.results[0][0].transcript; btn.innerText = currentAddress; document.getElementById('address-input').value = currentAddress; }; recognition.onerror = function() { btn.innerText = "Αποτυχία. Δοκιμάστε ξανά"; }; recognition.start(); } // Έναρξη Πλοήγησης και Αποθήκευση Δεδομένων function startNavigation() { if (document.getElementById('keys-ui').style.display === 'block') { currentAddress = document.getElementById('address-input').value; } if (!currentAddress || currentAddress.trim() === "") { alert("Παρακαλώ εισάγετε ή πείτε μια διεύθυνση πρώτα."); return; } // Συλλογή των δεδομένων από τη φόρμα const card = document.getElementById('card-amount').value; const tips = document.getElementById('tips-amount').value; const cash = document.getElementById('cash-amount').value; const status = document.querySelector('input[name="status"]:checked').value; // hand ή end // Δημιουργία αντικειμένου παραγγελίας const orderData = { id: Date.now(), // Μοναδικό ID με βάση την ώρα address: currentAddress, card: parseFloat(card), tips: parseFloat(tips), cash: parseFloat(cash), status: status === 'hand' ? 'Χέρι' : 'Τέλος', time: new Date().toLocaleTimeString('el-GR', { hour: '2-digit', minute: '2-digit' }) }; // Φόρτωση παλιών παραγγελιών και προσθήκη της νέας let history = JSON.parse(localStorage.getItem('delivery_history')) || []; history.push(orderData); localStorage.setItem('delivery_history', JSON.stringify(history)); // Άνοιγμα Google Maps App const encodedAddress = encodeURIComponent(currentAddress); window.location.href = `google.navigation:q=${encodedAddress}`; }
