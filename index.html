function renderApp() { const activeList = document.getElementById('active-orders-list'); const historyList = document.getElementById('history-list'); if (activeList) activeList.innerHTML = ''; if (historyList) historyList.innerHTML = ''; let cashTotal = 0; let cardTotal = 0; let tipsXerato = 0; let tipsTelos = 0; let deliveredCount = 0; for (let i = orders.length - 1; i >= 0; i--) { const order = orders[i]; if (order.status === 'done') { deliveredCount++; if (order.payment === 'Μετρητά') cashTotal += order.price; else cardTotal += order.price; if (order.tipType === 'Χεράτο') tipsXerato += order.tip; if (order.tipType === 'Στο Τέλος') tipsTelos += order.tip; } if (order.status === 'active') { if (activeList) { const card = document.createElement('div'); card.className = 'order-card'; // 🤖 Android Intent Link για εξαναγκαστικό άνοιγμα της εφαρμογής Google Maps const androidMapsUrl = `intent://://google.com{encodeURIComponent(order.address)}#Intent;scheme=http;package=com.google.android.apps.maps;end;`; card.innerHTML = `
${order.address}
${order.payment}
💰 Ποσό: ${order.price.toFixed(2)} €
🗺️ Χάρτης
`; activeList.appendChild(card); } } else { if (historyList) { const item = document.createElement('div'); item.className = 'history-card'; item.style.borderLeftColor = order.status === 'done' ? '#48bb78' : '#e53e3e'; item.innerHTML = `
${order.address} (${order.payment})
Ώρα: ${order.timestamp} ${order.tip > 0 ? `| Tip: ${order.tip.toFixed(2)}€ (${order.tipType})` : ''}
${order.status === 'done' ? '✔️' : '❌'} ${order.price.toFixed(2)} €
`; historyList.appendChild(item); } } } const tCount = document.getElementById('total-orders-count'); const tc = document.getElementById('total-cash'); const tcards = document.getElementById('total-cards'); const tx = document.getElementById('total-tips-xerato'); const tt = document.getElementById('total-tips-telos'); const tall = document.getElementById('total-all'); if (tCount) tCount.innerText = deliveredCount; if (tc) tc.innerText = `${cashTotal.toFixed(2)} €`; if (tcards) tcards.innerText = `${cardTotal.toFixed(2)} €`; if (tx) tx.innerText = `${tipsXerato.toFixed(2)} €`; if (tt) tt.innerText = `${tipsTelos.toFixed(2)} €`; if (tall) tall.innerText = `${(cashTotal - tipsTelos).toFixed(2)} €`; }
